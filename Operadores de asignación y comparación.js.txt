// Declaración de las variables
let numero1 = 10;
let numero2 = 5;
let numero3 = 15;

// 1. Muestre por consola si el numero1 es mayor o igual que numero2
console.log("¿numero1 es mayor o igual que numero2?", numero1 >= numero2);

// 2. Muestre por consola si el numero1 es menor o igual que numero2
console.log("¿numero1 es menor o igual que numero2?", numero1 <= numero2);

// 3. Muestre por consola si el numero1 es menor que numero3
console.log("¿numero1 es menor que numero3?", numero1 < numero3);

// 4. Muestre por consola si el numero3 es menor que numero2
console.log("¿numero3 es menor que numero2?", numero3 < numero2);

// 5. Muestre por consola si el numero3 es estrictamente diferente que numero1
console.log("¿numero3 es estrictamente diferente que numero1?", numero3 !== numero1);

// 6. Muestre por consola si el numero1 es estrictamente igual al numero2
console.log("¿numero1 es estrictamente igual al numero2?", numero1 === numero2);
